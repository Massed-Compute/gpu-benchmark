#!/usr/bin/env python3
"""mc-bench: Massed Compute GPU benchmark orchestrator (vLLM + SGLang)."""
from __future__ import annotations

import argparse
import json
import os
import sys
from pathlib import Path

ROOT = Path(__file__).resolve().parents[1]
sys.path.insert(0, str(ROOT))

from mc_bench import plan as plan_mod
from mc_bench import report as report_mod
from mc_bench import run as run_mod


def main() -> int:
    p = argparse.ArgumentParser(prog="mc-bench")
    sub = p.add_subparsers(dest="cmd", required=True)

    sp = sub.add_parser("plan", help="Pick SKUs for a model under spend caps")
    sp.add_argument("--model", required=True)
    sp.add_argument("--params-b", type=float, default=None)
    sp.add_argument("--include-l40s", action="store_true")

    sr = sub.add_parser("run", help="Launch → setup → bench → stats → terminate")
    sr.add_argument("--model", required=True)
    sr.add_argument("--params-b", type=float, default=None)
    sr.add_argument("--skus", default=None, help="Comma SKUs; default from plan")
    sr.add_argument("--include-l40s", action="store_true")
    sr.add_argument("--engines", default="vllm,sglang")
    sr.add_argument("--keep", action="store_true", help="Do not terminate after run")
    sr.add_argument("--dry-run", action="store_true")

    ss = sub.add_parser("report", help="Regenerate writeup from JSON")
    ss.add_argument("--model-slug", required=True)

    st = sub.add_parser("terminate", help="Terminate UUIDs from a run state file")
    st.add_argument("--state", required=True)

    args = p.parse_args()
    os.chdir(ROOT)

    if args.cmd == "plan":
        out = plan_mod.plan_skus(args.model, args.params_b, args.include_l40s)
        print(json.dumps(out, indent=2))
        return 0
    if args.cmd == "run":
        return run_mod.run_benchmark(args)
    if args.cmd == "report":
        path = report_mod.render_writeup(args.model_slug)
        print(f"Wrote {path}")
        return 0
    if args.cmd == "terminate":
        return run_mod.terminate_from_state(Path(args.state))
    return 1


if __name__ == "__main__":
    raise SystemExit(main())
